\section*{20. Základní úlohy o elipse 1}

\begin{ulohav}
Následující rovnice jsou \uv{zamaskované} rovnice (rovnoosých) elips; převeďte je do středového (tj. \uv{běžného}) tvaru, určete souřadnice středu, délky poloos, excentricitu a souřadnice ohnisek.
\begin{enumerate}
    \item $x^2+2 x+4 y^2-16 y+13 = 0$\lv{$\frac{(x+1)^2}{4} + \frac{(y-2)^2}{1} = 1$; střed $[-1;2]$, hlavní poloosa $2$, vedlejší poloosa $1$, excentricita $\sqrt3$, ohniska $[-1 - \sqrt3;2]$ a $[-1 + \sqrt3; 2]$}
    \item $16 x^2-96 x+5 y^2-10 y+69 = 0$\lv{$\frac{(x-3)^2}{5} + \frac{(y-1)^2}{16} = 1$; střed $[3; 1]$, hlavní poloosa $4$, vedlejší poloosa $\sqrt5$, excentricita $\sqrt{11}$, ohniska $[3; 1-\sqrt{11}]$ a $[3; 1 + \sqrt{11}]$}
\end{enumerate}
\end{ulohav}


\begin{uloha}
Napište (středovou) rovnici elipsy, jejíž osy splývají s osami souřadnic a která prochází body $K[2\sqrt3; \sqrt6]$ a $L[6;0]$.\vysl{$\frac{x^2}{36} + \frac{y^2}{9} = 1$}
\end{uloha}

\begin{ulohav}
Elipsa je dána rovnicí $4x^2 + 9y^2 = 36$. Najděte sjejí společné body s přímkou o rovnici
\begin{enumerate*}
    \item $2x + 3y - 6 = 0$,\lv{$[3;0]$ a $[0;2]$}
    \item $x - y - 6 = 0$,\lv{nemá průsečíky}
    \item $2x + 3\sqrt3 y - 12 = 0$.\lv{$[\frac32; \sqrt3]$}
\end{enumerate*}
\end{ulohav}

\end{uloha}
