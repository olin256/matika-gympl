\begin{ulohav}
Nalezněte chyby v následujících úpravách a určete správný výsledek:
\begin{enumerate}[itemsep=\medskipamount]
    \item $\frac{2^4 \cdot 3^3}{6^2} = \frac{(2 \cdot3)^{4+3}}{6^2} = \frac{6^7}{6^2} = 6^5$\lv{hned v prvním kroku $2^4 3^3 \neq (2 \cdot3)^{4+3}$; správně vyjde $2^2 \cdot 3 = 12$}
    \item $\left(\frac{x^2  y^5}{z}\right)^3 = \frac{x^6 y^{15}}{z}$\lv{neumocnil se jmenovatel; správně vyjde $\frac{x^6 y^{15}}{z^3}$}
    \item $\left(\frac{2^2 \cdot 3}{2^5 \cdot 3^4}\right)^2 = \frac{2^4 \cdot 3^2}{2^7 \cdot 3^6} = \frac{1}{2^3 \cdot 3^4}$\lv{špatně umocněný jmenovatel na druhou; správně vyjde $\frac{1}{2^6 \cdot 3^6}$}
    \item $\frac{3(a^2b^2)^3}{4(ab^2)^4} = \frac{3(a^6b^6)}{4(a^4b^8)} = \frac{3a^6 \cdot 3b^6}{4a^4 \cdot 4b^8} = \frac{9a^2}{16b^2}$\lv{$3(a^6b^6) \neq 3a^6 \cdot 3b^6$, není nic jako \uv{roznásobování}, pokud se v závorce násobí; správně vyjde $\frac{3a^2}{4b^2}$}
    \item $(-2)^{-2} = - \frac{1}{2^2} = -\frac14$\lv{při umocnění na sudou mocninu (i zápornou) se mínus \uv{nedostane ven}; správně vyjde $\frac{1}{4}$}
    \item $\frac{(3-5)^3}{7^2} = \frac{3^3 - 5^3}{7^2} = \frac{27-125}{49} = \frac{-98}{49} = -2$\lv{$(3-5)^3 \neq 3^3 - 5^3$, žádné takové pravidlo neexistuje; správně vyjde $-\frac{2^3}{7^2}$}
    \item $\left(\frac{a}{b}\right)^2 : \left(\frac{a}{b}\right)^{-2} = \frac{a^2}{b^2} : \frac{b^2}{a^2} = \frac{a^2 : a^2}{b^2 : b^2} = \frac{1}{1} = 1$\lv{v druhém kroce se špatně vydělily zlomky, mělo to být $\frac{\frac{a^2}{b^2}}{\frac{b^2}{a^2}} = \frac{a^2}{b^2} \cdot \frac{a^2}{b^2} = \frac{a^4}{b^4}$}
    \item $2^{-2} + 3^{-2} = \frac{1}{2^2 + 3^2} = \frac{1}{13}$\lv{správně to je $\frac{1}{2^2} + \frac{1}{3^2} = \frac{13}{36}$}
\end{enumerate}
\end{ulohav}
