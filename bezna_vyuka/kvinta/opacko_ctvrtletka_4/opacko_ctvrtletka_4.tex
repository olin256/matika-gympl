\section*{Opáčko před čtvrtletkou 4}


\begin{ulohav}
Vyřešte následující rovnice/nerovnice/soustavy... $x$, $y$ jsou vždy neznámé, ostatní písmena jsou parametry.
\begin{multicols}{2}
\begin{enumerate}
    \item $|x+1|+|2-x|-|x+3|=4$\lv{$\{-2;8\}$}
    \item $x^2 - |4x-1| + 3 = 0$\lv{$\{2;-2\pm\sqrt2\}$}
    \item $|x^2-3x+2| = 2x-3$\lv{$\left\{\frac12(1+\sqrt5); \frac12(5+\sqrt5)\right\}$}
    \item $|x^2 + 6| = 5x$\lv{$\{2;3\}$}
    \item $|3x+1| - 2x > 3$\lv{$(-\infty;-\frac45)\cup(2;\infty)$}
    \item $|x-2|\leq 2x$\lv{$\bigl\langle\frac23;\infty\bigr)$}
    \item $3+\sqrt{x-1} = x$\lv{$\{5\}$}
    \item $3\sqrt{x-6} = 4-x$\lv{$\emptyset$}
    \item $\sqrt{5-x^2} = x-1$\lv{$\{2\}$}
    \item $\sqrt{x} + 2\sqrt{10-x} = \sqrt{x+16}$\lv{$\{9\}$}
    \stitem $\sqrt{x^2-9}+x^2 = 21$\lv{$\{\pm3\sqrt2\}$}
    \item $\sqrt{2x-1} <x-2$\lv{$(5;\infty)$}
    \item $\sqrt{6-x} < 3x-4$\lv{$(2;6\rangle$}
    \item $\sqrt{2x+14} > x+3$\lv{$\langle-7;1)$}
    \item $6(2+x) = ax$\lv{$a \neq 6$: $\bigl\{\frac{12}{a-6}\bigr\}$,\quad $a = 6$: $\emptyset$}
    \item $(a^2-1)x = 2a^2 + a - 3$\lv{$a\neq\pm1$:~$\bigl\{\frac{2a+3}{a+1}\bigr\}$,\quad $a=1$:~$\R$,\quad $a=-1$:~$\emptyset$}
    \item $\frac{(a+1)^2}{4} = a(1-x+ax)$\lv{$a \neq 0;1$:~$\bigl\{\frac{a-1}{4a}\bigr\}$,\quad $a=0$: $\emptyset$,\quad $a=1$:~$\R$}
    \item $a^2-\frac1x = a\left(\frac1x - 1\right)$\lv{$a \neq 0;-1$:~$\bigl\{\frac1a\bigr\}$,\quad $a=0$:~$\emptyset$,\quad $a=-1$: $\R\setminus\bigl\{0\bigr\}$}
    \item $\frac{a(x+2)-3(x-1)}{x+1} = 1$\lv{$a\neq4;-6$:~$\bigl\{\frac{2a+2}{4-a}\bigr\}$,\quad $a=4$:~$\emptyset$,\quad $a=-6$:~$\emptyset$}
\end{enumerate}
\end{multicols}
\end{ulohav}
