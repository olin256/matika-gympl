    \textbf{\LARGE (Komplikovaná) sbírka}\\[.5mm]
    na\\[.5mm]
    {\Large lineární rovnice s parametrem}
\end{center}
Ve všech úlohách vystupuje $x$ jako neznámá, ostatní písmena označují parametry.

\begin{multicols}{2}
\everymath{\displaystyle}
\begin{enumerate}
    \rightskip0pt plus1fil\relax
    \lineskip=4pt\relax
    \pr{x(2a+1) = 5}{a = -\frac12 & \emptyset \\ a \neq -\frac12 & \bigl\{\frac{5}{2a+1}\bigr\}}
    \pr{t(3+t)x = 2t}{t=0 & \R \\ t=-3 & \emptyset \\ t \neq 0; -3 & \bigl\{\frac2{3+t}\bigr\}}
    \pr{x(p^2-1) = p^2+p}{p=-1 & \R \\ p=1 & \emptyset \\ p \neq \pm1 & \bigl\{\frac{p}{p-1}\bigr\}}
    \pr{x(a-4) = a^2-16}{a = 4 & \R \\ a \neq 4 & \{a+4\}}
    \pr{p(xp-1) = 1-x}{p \in \R & \bigl\{\frac{p+1}{p^2+1}\bigr\}}
    \pr{xa^2 = a(1+3x) - 3}{a = 3 & \R \\ a = 0 & \emptyset \\ a \neq 0;3 & \bigl\{\frac{1}{a}\bigr\}}
    \pr{s(2+x) = s^2 x - 6(x+1)}{s = -2; 3 & \emptyset \\ s \neq -2; 3 & \bigl\{\frac{2s+6}{s^2 - s - 6}\bigr\}}
    \prother{Určete všechny hodnoty parametru $p \in \R$ tak, aby množina řešení rovnice $2p(xp+1)-(p^2+1)x = 2$ obsahovala kladné číslo.}{\item $p \in (-\infty; -1) \cup \{1\}$}
    \prother{V rovnici $\displaystyle\frac mx + \frac{m+3}{2} = 8+\frac 1x$ určete hodnotu parametru $m \in \R$ tak, aby kořenem dané rovnice bylo číslo $2$.}{\item $m=7$}
    \pr{\frac{t}{x-2} = \frac{2}{x-t}}{t=2 & \R\setminus\{2\} \\ t=0 & \emptyset \\ t\neq 2;0 & \{t+2\}}
    \pr{\frac{2m}{2+x} = \frac{m-1}{x+1-m}}{m=-1 & \R\setminus\{-2\} \\ m=0;1 & \emptyset \\ m \neq 0;\pm 1 & \{2m-2\}}
    \pr{\frac{2x+m}{x+1} - \frac{3m}{x-m} = 2}{m=-1 & \R\setminus\{-1\} \\ m=0;2 & \emptyset \\ m \neq -1; 0; 2 & \{-\frac12m\}}
    \pr{\frac{3x}{x+2} = \frac{t}{t-1}}{t=1;\frac32 & \emptyset \\ t\neq 1;\frac32 & \bigl\{\frac{2t}{2t-3}\bigr\}}
    \pr{\frac{(2x-1)(2t-2)}{x-1} = t+1}{t=1;\frac53 & \emptyset \\ t\neq 1;\frac53 & \bigl\{\frac{t-3}{3t-5}\bigr\}}
    \pri{\sqrt{x^2 + 2p} = x + p}{p=0 & \R^+_0 \\ p<-2 & \emptyset \\ p\geq-2 \wedge p \neq 0 & \{1-\frac12p\}}
    \pri{|x-5| = k}{k < 0 & \emptyset \\ k \geq 0 & \{5-k;5+k\}}
    \pri{|x+5-k| = |x-2|}{k=7 & \R \\ k\neq7 & \{\frac12(k-3)\}}
\end{enumerate}
\end{multicols}
