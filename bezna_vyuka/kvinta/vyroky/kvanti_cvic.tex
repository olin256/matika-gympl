\section*{Výroky s kvantifikátory}

\let\AA\forall
\let\EE\exists

\begin{ulohav}
Rozhodněte o pravdivosti následujících výroků; pokud je výrok existenční a pravdivý, tak nalezněte příklad, pokud je obecný a nepravdivý, nalezněte protipříklad.
\begin{enumerate}
    \item $\EE x \in \Z \colon x = 10$\lv{1 $x = 10$}
    \item $\AA x \in \Z \colon x = 10$\lv{0 např. $x=11$}
    \item $\EE x \in \Z \colon (x = 10) \aand (x = 11)$\lv0
    \item $\EE x \in \Z \colon (x = 10) \oor (x = 11)$\lv{1 např. $x = 10$}
    \item $\EE x \in \R \colon x < 0$\lv{1 např. $x = -1$}
    \item $\EE x \in \N \colon x < 0$\lv0
    \item $\AA x \in \R \colon x^3 \geq 0$\lv{0 např. $x = -1$}
    \item $\EE x \in \R \colon (x < 0) \aand (x^2 > 0)$\lv{1 např. $x = -1$}
    \item $\AA x \in \R \colon (x < 0) \aand (x^2 > 0)$\lv{0 např. $x = 1$}
    \item $\AA x \in \R \colon (x \geq 0) \equ (x^3 \geq 0)$\lv{1}
    \item $\AA x \in \R \colon (x \geq 2) \equ (x^3 \geq 2)$\lv{0 např. $x = 1{,}5$}
    \item $\AA x \in \R \colon (x \geq 2) \imp (x^3 \geq 2)$\lv{1}
    %%%%%
    %%%%%
    \item $\EE x, y \in \R \colon x = y$\lv{1 např. $x = y = 0$}
    \item $\AA x, y \in \R \colon x = y$\lv{0 např. $x = 0$, $y = 1$}
    \item $\AA x, y \in \R \colon x^2 + y^2 \geq 0$\lv1
    \item $\EE x, y \in \R \colon x^2 + y^2 \leq 0$\lv{1 $x = y = 0$}
    \item $\EE x, y \in \R \colon x^2 + y^2 \geq 1$\lv{1 $x = y = 1$}
    \item $\AA x, y, z \in \R \colon (x \geq y) \imp (x + z \geq y + z) $\lv{1}
    \item $\AA x, y, z \in \R \colon (x \geq y) \imp (x \cdot z \geq y \cdot z)$\lv{0 např. $x =2$, $y = 1$, $z = -1$}
    \item $\AA x, y, z \in \R \colon ((x \geq y) \aand (y \geq z)) \imp (x \geq z)$\lv1
    \item $\AA x, y, z \in \R \colon ((x \geq y) \aand (y \geq z)) \equ (x \geq z)$\lv{0 např. $x=1$, $y=2$, $z = 0$}
    \item $\AA x \in \R\ \EE y \in \R \colon x > y$\lv1
    \item $\EE y \in \R\ \AA x \in \R \colon x > y$\lv0
\end{enumerate}
\end{ulohav}
