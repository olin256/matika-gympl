\begin{ulohav}
\everymath{\displaystyle}

\begin{multicols}{2}
\begin{enumerate}
    \item $\frac1x \geq x$\lv{$(-\infty;-1\rangle\cup (0;1\rangle$}
    \item $1-\frac{2-x}{1+x} > 0$\lv{$(-\infty;-1)\cup \bigl(\frac12;\infty\bigr)$}
    \item $\frac{x-2}{x-3} \leq \frac{x-5}{x-7}$\lv{$\langle-1;3) \cup (7;\infty)$}
    \item $\frac{5-x}{x-3} \leq \frac{x-5}{x-7}$\lv{$(-\infty;3)\cup \{5\} \cup (7;\infty)$}
    \item $\frac{6-4x}{x-2} < \frac{x}{2-x}$\lv{$\R \setminus\{2\}$ neboli $(-\infty;2)\cup(2;\infty)$}
    \item $\frac{1}{2x-1} + \frac{3}{2x+3} \geq 1$\lv{$\blz-\frac32;-\frac12\bra \cup \blz\frac12;\frac32\bra$}
    \item $\frac{1}{x^3-x} \geq \frac{1}{x^2-1}$\lv{$(-1;0)$}
    \item $\frac{1}{\frac{1}{x-1} + \frac{1}{x-2}} > 0$\lv{$\blz1;\frac32\brz \cup (2;\infty)$}
    \item $\frac{1}{\frac{1}{x-1} + \frac{1}{x-2}} \geq x$\lv{$\blz-\infty;-\sqrt2\bra \cup \bla\sqrt2;\frac32\brz$}
\end{enumerate}
\end{multicols}
\end{ulohav}
