\section*{Procvičování podílového tvaru}

\begin{ulohav}
\everymath{\displaystyle}
Vyřešte nerovnice:
\begin{enumerate}
    \item $\frac{1}{x-1} > 2$\lv{$\bigl(1; \frac32\bigr)$}
    \item $\frac{1}{x+1} > \frac{1}{x-1}$\lv{$(-1;1)$}
    \item $\frac{x-1}{x+1} \geq \frac{x+1}{x-1}$\lv{$(-\infty;-1) \cup \langle0;1)$}
    \item $\frac{1}{x^2-1} \leq \frac18$\lv{$(-\infty;-3\rangle\cup (-1;1) \cup \langle3; \infty)$}
    \item $1 + \frac{1}{x-1} \geq \frac{8}{x+2}$\lv{$(-\infty; -2)\cup (1;2\rangle \cup \langle4; \infty)$}
    \item $\frac{20}{x+3} > 7-2x$\lv{$\bigl(-3;-\frac12\bigr) \cup (1;\infty)$}
    \item $\frac{15-2x}{7-x}\leq \frac{1}{7-x}$\lv{$\emptyset$}
\end{enumerate}
\end{ulohav}
