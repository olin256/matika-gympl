\section*{9,5 Prostě logaritmy}


\begin{ulohav}
Určete \emph{bez kalkulačky} či jiných strojů následující čísla (strojově si můžete pak zkontrolovat, že jsou správně):
\everymath{\displaystyle}
\begin{multicols}{4}
\begin{enumerate}
    \item $\log_2 2$\lv{1}
    \item $\log_2 1$\lv{0}
    \item $\log_2 0$\lv{nedef.}
    \item $\log_2 16$\lv{4}
    \item $\log_2 \frac14$\lv{$-2$}
    \item $\log_2 0{,}25$\lv{$-2$}
    \item $\log_2 \sqrt[5]2$\lv{$\frac15$}
    \item $\log_{(-2)}2$\lv{nedef.}
    \item $\log_{(-2)}(-2)$\lv{nedef.}
    \item $\log_4 2$\lv{$\frac12$}
    \item $\log_4 1$\lv{$0$}
    \item $\log_4 8$\lv{$\frac32$}
    \item $\log_8 4$\lv{$\frac23$}
    \item $\log_{\frac12} 4$\lv{$-2$}
    \item $\log_{\frac18} 64$\lv{$-2$}
    \item $\log_{\frac12} (-1)$\lv{nedef.}
    \item $\log_{2} 2^{2021}$\lv{$2021$}
    \item $\log_{2} 4^{2021}$\lv{$4042$}
    \item $\log_{2} 2^{-2021}$\lv{$-2021$}
    \item $\log_{2^{43}} 2^{2021}$\lv{$47$}
    \item $\log_{2} (2^{7}\cdot 2^{12})$\lv{$19$}
    \item $\log_{2} \frac{2^7}{2^{12}}$\lv{$-5$}
    \item $\log_2 \frac1{2^\pi}$\lv{$-\pi$}
    \item $\log_{\sqrt 2}2$\lv{$2$}
    \item $\log_{\sqrt[3]2} \sqrt2$\lv{$\frac32$}
    \item $\log_{4^{13}}8^{5}$\lv{$\frac{15}{26}$}
\end{enumerate}
\end{multicols}
\end{ulohav}
