\section*{9. Exponenciální rovnice s \uv{hezkými} výsledky}


\def\lm#1{\lv{$\{#1\}$}}

\begin{ulohav}
Nalezněte všechna řešení těchto velmi zajímavých rovnic:
\everymath{\displaystyle}
\vskip-1.5\bigskipamount
\vbox{}
\begin{multicols}{2}
\raggedright
\begin{enumerate}
    \item $\sqrt5 \cdot 25^{2x-1} = 125^x$\lv{$\{\frac32\}$}
    \item $\left(\frac{\sqrt{7}}{\sqrt[3]{7}}\right)^{x} = \frac{\sqrt7}{49^x \cdot 7^x}$\lv{$\{\frac{3}{19}\}$}
    \item $4^x = 2^{x^2}$\lv{$\{0;2\}$}
    \item $5^x \cdot 2^x = 100^{x-1}$\lv{$\{2\}$}
    \item $3^x + 3^{x+1} = 108$\lm{3}
    \item $2^{x+1} + 2^{x-1} + 2^{x+3} = \frac{21}8$\lm{-2}
    \item $7 \cdot 4^{-x+2} = 3 \cdot 4^{-x+3} - 5$\lm{2}
    \item $3^x \cdot \left(\tfrac12\right)^x + 3^{x+1} \cdot \left(\tfrac12\right)^{x+1} = \tfrac53$\lm{-1}
    \item $4^{2x} - 6 \cdot 4^x + 8 = 0$\lm{\frac12; 2}
    \item $\frac14 \cdot 2^x + \frac12 \cdot 4^x = 9$\lm{2}
    \item $2 \cdot \left(\tfrac14\right)^x - 3 \cdot \left(\tfrac12\right)^x = \left(1 + \left(\tfrac12\right)^x\right)\cdot\left(\tfrac14\right)^{-1}$\lm{-2}
    \item $9\cdot 3^{x} + 3^{-x} = 10$\lm{-2; 0}
    \item $8^{2 x + 1} + 4 = 8^x + 8^{x + \frac53}$\lm{-1; \frac23}
    \stitem $256 \cdot 2^{4^x} = 64^{2^x}$\lm{1; 2}
    \stitem $3^x + 3^{x+1} = 7 \cdot 4^x - 4^{x+1}$\lm{1}
    \stitem $2^{x-1} - 2^{x-2} = 5^{x-3} + 2^{x-3}$\lm{3}
    \stitem $4^x + 6^x = 2 \cdot 9^x$\lm{0}
    \stitem $16^x = 8 \cdot 4^x + 2 \cdot 8^x$\lm{2}
\end{enumerate}
\end{multicols}
\end{ulohav}
